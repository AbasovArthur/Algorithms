``````````````````````````````````````````````````` function binarySearch(arr, find) { let left = 0; let right = arr.length - 1; while (left <= right) { let middle = Math.floor((left + right) / 2); if (arr[middle] == find) { return middle; } else if (arr[middle] < find) { left = middle + 1; } else { right = middle - 1; } } return -1; } // Пример использования функции binarySearch let exampleArray = [1, 3, 5, 7, 9, 11, 13, 15]; let findElement = 15; let result = binarySearch(exampleArray, findElement); console.log("Индекс искомого элемента:", result);
